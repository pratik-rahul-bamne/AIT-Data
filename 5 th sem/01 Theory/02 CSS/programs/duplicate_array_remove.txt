<script>
let arr = [1, 2, 3, 4, 5, 6, 7, 8, 9, 1, 2, 3, 4, 5];
let uniqueArr = [];

for (let i = 0; i < arr.length; i++) {
  if (uniqueArr.indexOf(arr[i]) === -1) {
    uniqueArr.push(arr[i]);
  }
}

document.write("Original Array: " + arr + "<br>");
document.write("Array with Duplicate Elements Removed: " + uniqueArr);
</script>
